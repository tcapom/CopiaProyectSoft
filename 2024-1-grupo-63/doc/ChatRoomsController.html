def authorize_user!

def create
  @chat_room = ChatRoom.new(chat_room_params)
  if @chat_room.save
    ChatRoomUser.create(chat_room: @chat_room, user: current_user)
    redirect_to @chat_room
  else
    render :new
  end
end
